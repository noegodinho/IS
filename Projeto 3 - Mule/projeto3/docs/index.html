<?xml version="1.0" encoding="UTF-8"?>

<flow name="Subscribe">
    <set-variable variableName="email" value="#[payload[0]]" doc:name="SetEmail">   </set-variable>
    <set-variable variableName="course" value="#[payload[1]]" doc:name="SetCourse"> </set-variable>
    <logger message="#[email] #[course]" level="INFO" doc:name="Logger">    </logger>
    <db:select config-ref="MySQL_Configuration" doc:name="Database">
        <db:parameterized-query><![CDATA[SELECT email, course FROM Subscription WHERE email=#[email] AND course=#[course];]]>   </db:parameterized-query>
    </db:select>
    <choice doc:name="Choice">
        <when expression="#[payload.isEmpty()]">
            <set-payload value="You are being subscribed to the Course &quot;#[course]&quot; Notification List. Click http://localhost:8081/checkSubscription?email=#[email]&amp;course=#[course] to accept " doc:name="Write Email"/>
            <smtp:outbound-endpoint host="smtp.gmail.com" port="587" user="isproject3mule%40gmail.com" password="isisisis" connector-ref="Gmail" to="#[email]" from="isproject3mule%40gmail.com" subject="Subscribe confirmation" responseTimeout="10000" doc:name="Send Confirmation Email">   </smtp:outbound-endpoint>
            <db:update config-ref="MySQL_Configuration" doc:name="Database">
                <db:parameterized-query><![CDATA[UPDATE Statistics SET numberEmails = numberEmails+1, today = today+1;]]>   </db:parameterized-query>
            </db:update>
            <set-payload value="&quot;Confirmation email sent&quot;" doc:name="Set Payload"/>
        </when>
        <otherwise>
            <logger message="Pair already in database" level="INFO" doc:name="Logger"/>
            <set-payload value="&quot;Pair already in database&quot;" doc:name="Set Payload"/>
        </otherwise>
    </choice>
    <exception-strategy ref="LogExceptions" doc:name="Reference Exception Strategy">    </exception-strategy>
</flow>
